SVN Setup on Manaus
--------------------

1. yum install subversion mod_dav_svn
2. vi <APACHE_HOME>/conf.d/subversion.conf
	
	LoadModule dav_svn_module     modules/mod_dav_svn.so
	LoadModule authz_svn_module   modules/mod_authz_svn.so
	
	<VirtualHost *:80>
        	ServerName svn.conversant.co.nz
        	<Location /adempiere>
                	DAV svn
                	SVNPath /ebs/svn/repo/adempiere
                	AuthType Basic
                	AuthName "Conversant SVN"
                	AuthUserFile /ebs/svn/auth/adempiere
                	Require valid-user
	        </Location>
	</VirtualHost>

3. htpasswd -cm /ebs/svn/auth/adempiere josh.hill
4. svnadmin create /ebs/svn/repo/adempiere
5. chmod -R 777 /ebs/svn
6. chown -R asterisk:asterisk /ebs/svn
7. service httpd restart
